        static void AddPercent(DataTable table)
        {
            int t=table.Columns.Count;
            foreach (DataRow row in table.Rows)
            {
                for (int i = 1; i < t; i = i + 1)
                {
                    string columnName = table.Columns[i].ColumnName;
                    if (double.TryParse(row[columnName].ToString(), out double originalValue))
                    {
                        string percentageValue = originalValue.ToString("0.00") + "%";
                        row[columnName] = percentageValue;
                    }
                    else
                    {
                        Console.WriteLine($"无法解析值: {row[columnName]}");
                    }
                }
            }
        }
